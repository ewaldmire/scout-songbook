\songtitle{I Met A Bear}

From: \url{https://www.scoutsongs.com/lyrics/imetabear.html}

\begin{center}
\textit{Scouts repeat each line during 1st part of verses} \\
\end{center}

\begin{verse}
The other day (scouts repeat)\\
I met a bear (scouts repeat)\\
A great big bear (scouts repeat)\\
Oh way out there. (scouts repeat)\\
\end{verse}

\begin{center}
\textit{Same lines in unison}\\
\end{center}

\begin{verse}
The other day I met a bear!!\\
A great big bear Oh way out there!!\\
\end{verse}


\begin{center}
\textit{Sing additional verses similarly}\\
\end{center}

\begin{verse}
He looked at me, I looked at him,\\
He sized up me, I sized up him.\\
\end{verse}

\begin{verse}
He said to me, "Why don't you run?\\
I see you ain't, Got any gun."\\
\end{verse}

\begin{verse}
I says to him, "That's a good idea."\\
"Now legs get going, get me out of here!"\\
\end{verse}

\begin{verse}
And so I ran, Away from there,\\
But right behind me, Was that bear.\\
\end{verse}

\begin{verse}
In front of me, There was a tree,\\
A great big tree, Oh glory be!\\
\end{verse}

\begin{verse}
The lowest branch, Was ten feet up,\\
So I thought I'd jump, And trust my luck.\\
\end{verse}

\begin{verse}
And so I jumped, Into the air,\\
But I missed that branch, A way up there.\\
\end{verse}

\begin{verse}
Now don't you fret, And don't you frown,\\
I Caught that branch, On the way back down!\\
\end{verse}

\begin{verse}
This is the end, There aint no more,\\
Unless I see, That bear once more. 
\end{verse}
